客服（service）：
	id（序号，数字，主键，自增）	
	name（客服名称，字符，非空）
	store_id（店铺外键，）

create table service(id int primary key auto_increment comment '客服id',name char(10) not null comment '客服名称',foreign key(id) references store(id) on delete cascade on update cascade comment '店铺外键');

对话记录（chattingrecords）：
	id（序号，数字，主键，自增）	
	text（对话记录，文本，非空）
	time（对话时间，时间戳，非空）
	user_id（用户外键，默认值为false）
	service_id（客服外键，默认值为false）
	me_id（子关联外键，）
	# 子关联回复消息 # 关联自身id的回话

# 表关系写清楚，数据库创建语句，字段的作用表达明确，讲解用法

create table chattingrecords(id int primary key auto_increment comment '对话记录 id',text text not null comment '对话记录',time datetime not null comment '对话时间',foreign key(id) references user(id),foreign key(id) reference service(id),foreign key(me_id) reference chattingrecords(id));
